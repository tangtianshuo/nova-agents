```typescript
const validateForm = (): { valid: boolean; errors: Record<string, string> } => {
  const errors: Record<string, string> = {};

  if (!formData.type) {
    errors.type = '请选择传输协议';
  }
  if (!formData.id.trim()) {
    errors.id = 'ID 不能为空';
  }
  if (!formData.name.trim()) {
    errors.name = '名称不能为空';
  }
  if (formData.type === 'stdio' && !formData.command?.trim()) {
    errors.command = 'STDIO 类型需要命令';
  }

  return { valid: Object.keys(errors).length === 0, errors };
};
```

6. Implement JSON validation:
```typescript
const validateJson = (): { valid: boolean; errors: string; data?: any } => {
  try {
    const parsed = JSON.parse(mcpJsonInput);
    if (!parsed.mcpServers || typeof parsed.mcpServers !== 'object') {
      return { valid: false, errors: 'JSON 必须包含 mcpServers 对象' };
    }
    return { valid: true, errors: '', data: parsed };
  } catch (err) {
    return { valid: false, errors: 'JSON 格式无效' };
  }
};
```
