```tsx
function ToggleSwitch({ enabled, loading, onToggle }: { enabled: boolean; loading?: boolean; onToggle: () => void | Promise<void> }) {
  return (
    <button
      onClick={onToggle}
      disabled={loading}
      className={`relative h-6 w-11 shrink-0 rounded-full transition-colors ${
        loading
          ? 'bg-[var(--info)]/60 cursor-wait'
          : enabled
            ? 'cursor-pointer bg-[var(--accent)]'
            : 'cursor-pointer bg-[var(--line-strong)]'
      }`}
    >
      <span
        className={`absolute left-0.5 top-0.5 h-5 w-5 rounded-full bg-white shadow transition-transform ${enabled ? 'translate-x-5' : 'translate-x-0'}`}
      />
    </button>
  );
}
```
